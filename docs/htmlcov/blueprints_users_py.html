114 @users.response(203, 'user logged in', user_token) 

115 @users.response(403, 'invalid creadentials') 

116 @users.response(400, 'invalid request') 

117 @users.expect(user_and_pw) 

118 def post(self): 

119 ''' takes generates a user token if the user name 

120 and password are correct ''' 

121 req = users.payload 

122 try: 

123 # get user from database 

124 user = User.query.filter_by(username=req['username']).first() 

125 # tests to see if the provided password matches found user 

126 # if no user is found will raise a key error 

127 # if incorect password raises an attribute error 

128 if user is None or not crypt.verify(req['password'], 

129 user.password): 

130 return Response('invalid creadentials', status=400) 

131 # prepare responce create token 

132 data = {} 

133 data['user'] = user.to_dict() 

134 data['token'] = create_access_token(identity=user.to_dict()) 

135 response = jsonify(data) 

136 response.set_cookie('refresh_token', 

137 create_refresh_token(identity=user.to_dict()), 

138 httponly=True, 

139 secure=True) 

140 response.status_code = 203 

141 # return token and user info 

142 return response 

143 except (IntegrityError, KeyError): 

144 # if a database error or password error is encountered returns 400 

145 return Response('invalid request', status=400)
